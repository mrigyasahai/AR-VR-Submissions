import java.util.Scanner;
public class Innogeeks {


    public static void main(String[] args) throws Exception {
        int num,k;
        Scanner sc  = new Scanner(System.in);
         num = sc.nextInt();
        k = factorial(num);
        System.out.print(k);
    }

    public static int factorial(int n){
       if(n==0)
       return 1;
       else
       return(n*factorial(n-1));
        
    }

}
